Command routing, strongly-typed argument parsing, validation, and automatic usage help for console applications. Make console apps that don't suck. — Faster.

Getting started…

  • 1st Install the DotConsole package using NuGet
    or download it old school.
  • 2nd Create a command, add some parameters,
    and use the built-in DataAnnotations for validation.
    // todo: add highlighting

using System.ComponentModel.DataAnnotations;
using System.Net.NetworkInformation;
using DotConsole;
        
public void PingCommand : ICommand
{
  public IEnumerable CommandNames
  {
    get
    {
      return new string[] { "ping" };
    }
  }
  
  [Parameter(Name = "hostname", Position = 0)]
  [Required]
  public string HostName { get; set; }
  
  public void Execute()
  {
    var pinger = new Ping();
    var reply = pinger.Send(HostName);
    if (reply.Status == IPStatus.Success)
    {
      Console.WriteLine("Success!");
    }
    else
    {
      Console.WriteLine("No reply.");
    }
  }
}
